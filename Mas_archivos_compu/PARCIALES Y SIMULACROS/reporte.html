    res:bool=True

def subsecuencia_mas_larga(v: list[int]) -> tuple[int,int]:
    longitud = (len(v)-1)
    indice = 0
    cantidad = 0
    lista = []
    while indice <= longitud:
        if not todos_consecutivos(lista):
            lista = []
            cantidad = 0
        if  todos_consecutivos(lista) and longitud > indice:
            lista.append(v[indice])
            longitud -=1
            indice+=1
            cantidad = len(lista)

    return (cantidad,indice-1)
